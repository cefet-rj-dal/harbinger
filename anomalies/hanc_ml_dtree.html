# data preprocessing
slevels <- levels(dataset$event)

train <- dataset[1:80,]
test <- dataset[-(1:80),]

norm <- minmax()
norm <- fit(norm, train)
train_n <- transform(norm, train)

# establishing decision tree method 
model <- hanc_ml(cla_dtree("event", slevels))

# fitting the model
model <- fit(model, train_n)
detection <- detect(model, train_n)
